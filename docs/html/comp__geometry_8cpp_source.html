46 // Returns YES if the intersection point was found, and stores that point in X,Y.

50  double Cx, double Cy,double Dx, double Dy,
51  double &X, double &Y)
52 {
53 
54  double distAB, theCos, theSin, newX, ABpos ;
55 
56  // Fail if either line segment is zero-length.
57  if (Ax==Bx && Ay==By || Cx==Dx && Cy==Dy) return false;
58 
59  // Fail if the segments share an end-point.
60  if (Ax==Cx && Ay==Cy || Bx==Cx && By==Cy
61  || Ax==Dx && Ay==Dy || Bx==Dx && By==Dy) {
62  return false; }
63 
64  // (1) Translate the system so that point A is on the origin.
65  Bx-=Ax; By-=Ay;
66  Cx-=Ax; Cy-=Ay;
67  Dx-=Ax; Dy-=Ay;
68 
69  // Discover the length of segment A-B.
70  distAB=sqrt(Bx*Bx+By*By);
71 
72  // (2) Rotate the system so that point B is on the positive X axis.
73  theCos=Bx/distAB;
74  theSin=By/distAB;
75  newX=Cx*theCos+Cy*theSin;
76  Cy =Cy*theCos-Cx*theSin; Cx=newX;
77  newX=Dx*theCos+Dy*theSin;
78  Dy =Dy*theCos-Dx*theSin; Dx=newX;
79 
80  // Fail if segment C-D doesn't cross line A-B.
81  if (Cy<0. && Dy<0. || Cy>=0. && Dy>=0.) return false;
82 
83  // (3) Discover the position of the intersection point along line A-B.
84  ABpos=Dx+(Cx-Dx)*Dy/(Dy-Cy);
85 
86  // Fail if segment C-D crosses line A-B outside of segment A-B.
87  if (ABpos<0. || ABpos>distAB) return false;
88 
89  // (4) Apply the discovered position to line A-B in the original coordinate system.
90  X=Ax+ABpos*theCos;
91  Y=Ay+ABpos*theSin;
92 
93  // Success.
94  return true;
95 }
